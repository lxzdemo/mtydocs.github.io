# 安装

# npm 安装

推荐使用 npm 的方式安装，它能更好地和 webpack 打包工具配合使用。

npm i mty-ui

# 全局导入

// 在main.js导入
import mtyUI from 'mty-ui'
import 'mty-ui/dist/mty-ui.css'
Vue.use(mtyUI)

# 按需导入

// 在main.js导入
import { Button } from 'mty-ui'
import 'mty-ui/dist/mty-ui.css'
Vue.use(Button)
